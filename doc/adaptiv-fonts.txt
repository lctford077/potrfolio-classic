.title {

    // соотношение 1.33, т.е. 80 / 1.33 = 60
    // max 80px, min 60px
    // мы хотим что бы шрифт начинал уменьшаться например с 1000px,
    // 1000px / 100 = 10vw, 
    // max размер т.е 80px делим на 10vw и получаем 8vw.
    // примерно на 745px шрифт станет 60px и больше не уменьшится.


    font-size: clamp(60px, 8vw, 80px);
    letter-spacing: -1px;
    line-height: 1.2;
}
